
import React, { useState } from 'react';
import { JournalEntry, Mood } from '../types';
import { MOOD_OPTIONS } from '../constants';
import Icon from './Icon';

interface JournalEntryFormProps {
  addJournalEntry: (entry: Omit<JournalEntry, 'id' | 'date'>) => void;
}

const JournalEntryForm: React.FC<JournalEntryFormProps> = ({ addJournalEntry }) => {
  const [text, setText] = useState('');
  const [selectedMood, setSelectedMood] = useState<Mood>(Mood.Neutral);
  const [activities, setActivities] = useState('');

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (text.trim() === '') return;

    const activityArray = activities.split(',').map(a => a.trim()).filter(a => a !== '');
    
    addJournalEntry({ text, mood: selectedMood, activities: activityArray });

    setText('');
    setSelectedMood(Mood.Neutral);
    setActivities('');
  };

  return (
    <div className="bg-white dark:bg-slate-800 p-6 rounded-lg shadow-md">
      <h2 className="text-lg font-semibold mb-4 text-slate-700 dark:text-slate-200">How are you feeling today?</h2>
      <form onSubmit={handleSubmit} className="space-y-4">
        <div>
          <textarea
            value={text}
            onChange={(e) => setText(e.target.value)}
            placeholder="Share your thoughts..."
            className="w-full h-32 p-3 bg-slate-100 dark:bg-slate-700 border border-slate-200 dark:border-slate-600 rounded-md focus:ring-2 focus:ring-indigo-500 focus:border-indigo-500 transition-colors"
            required
          />
        </div>
        <div>
          <label className="block text-sm font-medium text-slate-600 dark:text-slate-300 mb-2">Select your mood:</label>
          <div className="flex flex-wrap gap-2">
            {MOOD_OPTIONS.map(({ mood, label, icon, color, darkColor }) => (
              <button
                key={mood}
                type="button"
                onClick={() => setSelectedMood(mood)}
                className={`flex items-center space-x-2 px-3 py-2 rounded-full text-sm font-medium border-2 transition-all duration-200 ${
                  selectedMood === mood
                    ? 'bg-indigo-500 border-indigo-500 text-white'
                    : `bg-white dark:bg-slate-700 border-slate-200 dark:border-slate-600 hover:border-indigo-400 dark:hover:border-indigo-500`
                }`}
              >
                <Icon name={icon} className={`w-5 h-5 ${selectedMood !== mood ? `${color} ${darkColor}` : ''}`} />
                <span>{label}</span>
              </button>
            ))}
          </div>
        </div>
        <div>
          <label htmlFor="activities" className="block text-sm font-medium text-slate-600 dark:text-slate-300 mb-2">
            Activities (comma-separated):
          </label>
          <input
            id="activities"
            type="text"
            value={activities}
            onChange={(e) => setActivities(e.target.value)}
            placeholder="e.g., work, exercise, friends"
            className="w-full p-2 bg-slate-100 dark:bg-slate-700 border border-slate-200 dark:border-slate-600 rounded-md focus:ring-2 focus:ring-indigo-500 focus:border-indigo-500 transition-colors"
          />
        </div>
        <div className="flex justify-end">
          <button
            type="submit"
            className="px-6 py-2 bg-indigo-600 text-white font-semibold rounded-md hover:bg-indigo-700 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-indigo-500 dark:focus:ring-offset-slate-800 disabled:opacity-50 transition-colors"
            disabled={text.trim().length === 0}
          >
            Save Entry
          </button>
        </div>
      </form>
    </div>
  );
};

export default JournalEntryForm;
